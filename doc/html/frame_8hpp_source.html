43 match_features(cv::Mat kp1, cv::Mat desc1, cv::Mat kp2, cv::Mat desc2, float ratio = 0.80){
44 std::vector<std::vector< cv::DMatch > > rawMatches;
45 //matcher.match(descriptors1, descriptors2, matches);
46 matcher.knnMatch(desc1, desc2, rawMatches, 2);
47 // perform Lowe's ratio test to get actual matches
48 std::vector<cv::DMatch> matches;
49 cv::Mat pts1;
50 cv::Mat pts2;
51 cv::Mat ft1;
52 cv::Mat ft2;
53 for(auto it = rawMatches.begin(); it != rawMatches.end(); it++){
54 if( (*it)[0].distance < ratio * (*it)[1].distance ){
55 pts1.push_back( kp1.row((*it)[0].queryIdx) );
56 pts2.push_back( kp2.row((*it)[0].trainIdx) );
57 ft1.push_back( desc1.row((*it)[0].queryIdx) );
58 ft2.push_back( desc2.row((*it)[0].trainIdx) );
59 matches.push_back((*it)[0]);
60 }
61 }
62 return std::tuple(matches, pts1, ft1, pts2, ft2);
63 }
