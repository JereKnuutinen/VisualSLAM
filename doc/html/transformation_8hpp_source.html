107 cv::Mat R; // Rotation
108 cv::Mat t; // translation

114 cv::cv2eigen(t, t_);
115 Eigen::MatrixXd pos = Isometry3d(R_, t_).matrix().inverse();
116 cv::eigen2cv(pos, T_);
117 triangulatedPoints = triangulated_points_cv.t(); // transpose and store to private member
118 }
